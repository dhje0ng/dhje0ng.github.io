Find My Location

  • Node.JS Prototype Pollution
const express = require('express');
const bodyParser = require('body-parser');
const cors = require('cors');
const app = express();
const cookieParser = require('cookie-parser');
app.use(cookieParser());
app.use(express.json())
app.set('view engine', 'ejs');
const port = process.env.PORT || 3000

app.use(express.static(__dirname + '/public'));

app.use(cors())

const users = [
    { name: 'user', password: 'pwd' },
    { name: 'admin', password: Math.random().toString(32), isAdmin: true },
];

const myLoaction = [
    { flag: "GLUG{f4k3_fl4g_f0r_t3st1ng}" }
]

function findUser(auth) {
    return users.find((u) =>
        u.name === auth.name &&
        u.password === auth.password);
}

app.use(bodyParser.json());


app.get('/', (req, res) => {

    historys = [];
    const data = historys

    res.render('index', { data });
});

app.post('/', (req, res) => {
    console.log(req.body)
    const user = findUser(req.body.auth || {});

    if (!user) {
        res.status(403).send({ ok: false, error: 'Access denied' });
        return;
    }

    const history = {
        icon: '👋',
    };

    Object.assign(history,req.body.location)

    if (history.isAdmin == true) {
        res.status(200).send(myLoaction)

    } else {
        res.status(200).send(history)
    }
})

app.listen(port);
console.log(`Listening on port ${port}`);

문제의 코드는 다음과 같다. 문제를 풀기 위해서는 어드민 계정이 필요할 것 같지만

코드를 조금 읽다보면 이상한 점을 하나 발견할 수 있다.

Object.assign(history,req.body.location)

if (history.isAdmin == true) {
    res.status(200).send(myLoaction)

} else {
    res.status(200).send(history)
}

플래그를 출력하는 로직이 history 객체에서 isAdmin=true 일 경우 플래그를 얻을 수 있는데

Object.assign() 통해서 history 객체에 post 데이터로 들어온 location 매개변수 값을 파싱해서 할당해주고 있다.

예를 들어서 "location": {"name": "test"} 이런식으로 데이터를 보내면 {"name": "test"} 값이

history 객체에 들어간다고 이해하면 된다. 이렇게 되면 그냥 history 객체에 isAdmin 오브젝트를 써주면 되서

prototype pollution 트리거가 가능하다.

Exploit

"location": {"name": "test", "isAdmin": true}
